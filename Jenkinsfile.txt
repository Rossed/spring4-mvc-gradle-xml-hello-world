pipeline {
    agent any

    parameters {
            string(name: 'Username', defaultValue: 'Ross', description: 'Who is running the build')
        }
    stages {    
        stage('Build') {
            steps {
                echo "Hello ${params.Username}, we are building $BUILD_TAG"
                pwd()
                sh './gradlew build'
            }
            post {
                success {
                    echo "$BUILD_TAG SUCCESSFULLY BUILT"
                }
                failure {
                    echo "FAILURE IN BUILD $BUILD_TAG"
                }
            }
        }
        stage('Deploy') {
            steps {
                parallel (
                    'test 1': {
                        echo "Deploying $BUILD_TAG..."
                        sh './gradlew run --info'
                    },
                    'test 2': {
                        echo 'running test 2'
                    }
                )
            }
            post {
                success {
                    echo "$BUILD_TAG SUCCESSFULLY DEPLOYED"
                }
                failure {
                    echo "FAILURE WHEN DEPLOYING BUILD $BUILD_TAG"
                }
            }
        }
        stage('Test') {
            steps {
                echo "Testing $BUILD_TAG..."
            }
            post {
                success {
                    echo "ALL TESTS FOR $BUILD_TAG SUCCEEDED"
                }
                failure {
                    echo "FAILURE WHEN TESTING $BUILD_TAG"
                }
            }
        }
    }
}
